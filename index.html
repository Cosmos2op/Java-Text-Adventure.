public class DungeonEscape {
    public static void main(String[] args) {
        Game game = new Game();
        game.start();

        while (game.isRunning()) {
            String command = System.console().readLine("> ");
            game.processCommand(command);
        }

        System.out.println("Game Over.");
    }
}
